def as_time(time)
  return nil unless time

  if time.is_a?(String)
    parse_time(time)
  elsif time.is_a?(ActiveSupport::TimeWithZone)
    time
  elsif time.respond_to?(:to_time)
    time.to_time
  else
    Array(time).flat_map { |d| as_time(d) }
  end
end

def days_in_month(month, year = current_time.year)
  date = ::Date.new(year, month, 1)
  ((date >> 1) - date).to_i
end
